前后端通信报文的设计

背景： 为工作流和LLM混合的系统设计前后端的通信格式。系统的核心功能是为了给处理用户寿险红利领取的请求。
整个领取流程相对固定，LLM的作用是帮助用户简化UI端操作 - 即用户既可以直接和UI元素交互，也可以通过文本输入或语音输入完成选择和提交动作。
由于整个客户端是一个对话界面，UI元素大部分是根据流程渲染出的一些卡片风格的表单。

v1版本设计
输入 - 从UI到后端， 与输出 - 后端返回UI的数据结构基本一致。 
Json 格式的数据结构如下，
{ 
// system identification 
 “session_id”:  用户登录后的会话ID,
 “context_id”:  用于记录当前办理流程上下文的ID，
 “token_id”: 登陆授权ID，
 “bu_id”: business unit ID,
 “accessEntrance”: 客户端App标识， 
// common properties
 “to_agent”: 不为空时，请求将会被gateway直接转发给特定Agent，
 “current_card”: 当前正在处理的卡片ID，
//  用于页面渲染的，或从前端提交给后端的数据结构
//  结构化数据
 “card”: {
   “name”: xx,
   “id”: xx,
   “card_type”: UI端以此识别如何进行渲染,
   “content_description”:xx, 
   “content”: [ {key: value} }
   sub_cards(optional): [
       card_1,
       card_2,
   ]
], 
“msg_input”: 非结构化的用户输入
“msg_response”: LLM response，
“msg_type”: 当前输入或输出的类型，1 - 文本输入/输出， 2- 结构化输入/输出， 3-语音信息，4..
}

V2版本 - 经办流程触发后是否有ID？
{
  // --- 系统级别信息 ---
  "session_id": "string",        // 用户登录后的会话ID
  "context_id": "string",        // 办理流程上下文ID，用于状态跟踪
  "token_id": "string",          // 登录授权ID / JWT， 由gateway自动添加？
  "bu_id": "string",             // Business Unit ID
  "accessEntrance": "string",    // 客户端入口标识（如app/web/h5/ivr）
  "version": “1.0”,                        // 报文版本，用于协议兼容性管理

  // --- 路由与状态控制 ---
  "to_agent": "string|null",     // 指定Agent时非空，否则由后端统一调度
  "current_card_id: "string|null", // 当前处理中的卡片ID（表单或组件） 
  "status": “string|null”, // 流程状态：success, pending, error, loading, confirmation_required, etc. 
  "error": { 
      // 可选：当 status 为 'error' 时，提供错误详情 
     "code": 4001, 
     "message": "保单信息不匹配，请重新输入。" },

  // --- UI渲染数据（结构化交互），在用户交互后在msg_input中拼出值对信息 ---
  "card": {
    "id": "string",
    "name": "string",
    "card_type": "form|list|info|confirm|custom", // 渲染类型
    "content_description": "string",              // UI展示的简要说明
    "content": [
      {
        "key": "string",
        "label": "string",
        "type": "string",       // input类型 (text, date, select, number)
        "value": "any",         // 用户输入值或默认值
        "required": true,
        "options": [              // dropdown list
          { "label": "string", "value": "string" }
        ]
      }
    ],
    "sub_cards": [ /* 嵌套子卡片 */ ]
  },

  // --- 用户与LLM交互 ---
  "msg_input": "string",           // 仅为输入结构，非结构化用户输入（文本/语音转文字）
  "msg_response": "string",    // 仅为输出结构，LLM 返回的文本响应（可包含UI操作建议）
  "msg_type": 1,                       // 1-文本, 2-结构化, 3-语音, 4-图片...

  // --- 元信息 ---
  "timestamp": "2025-09-10T14:23:11Z", // 请求/响应时间戳
  "trace_id": "string"                 // 日志与链路追踪用
}
